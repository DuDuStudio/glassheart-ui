使用代碼

如何在您的專案中使用 GlassHeartUI

<!-- 引入樣式 --> <link rel="stylesheet" href="https://unpkg.com/glassheart-ui-core@1.1.0/dist/index.css"> <!-- 引入 JavaScript --> <script src="https://unpkg.com/glassheart-ui@1.1.0/dist/index.js"></script> <script> const { GlassCard, GlassButton, GlassInput } = window.GlassHeartUI; // 創建卡片 const card = new GlassCard({ size: 'lg', liquid: true, interactive: true, content: '<h3>Hello World!</h3>' }); // 渲染到頁面 card.render('#app'); </script>
